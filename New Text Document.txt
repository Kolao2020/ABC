       a.vin_no, a.engine_no, a.mission_no, a.key_no,  a.job_no, a.job_dt, a.job_sq,
       mdl_bc=(select mdl_bc from ppp160 where model_mes=a.model_mes),
       color_bc=(select color_bc from ppp160 where model_mes=a.model_mes)
  from pdp170 a
  left join ppp160 b on a.model_mes=b.model_mes 
 where 1=1
   and set_line=@set_line
 andif a.vin_no like '%'+@vin_no endif
   and job_dt>'2020-06-01'

        or vin_no not in (select vin_no from PDP210 ) )
 order by a.job_dt, a.job_sq
